\subsection{Funding Paradox Decomposition}

\begin{equation}
\underbrace{\frac{dG}{dF} < 0}_{\text{Paradox}} = \underbrace{\frac{dG}{dM} > 0}_{\text{Movement Principle}} \times \underbrace{\frac{dM}{dF} < 0}_{\text{Funding Trap}}
\end{equation}

\subsection{Learning Trap Condition}

\begin{equation}
\text{Trap when: } \mu(1-\mu) < \frac{\varepsilon}{V+1}
\end{equation}
